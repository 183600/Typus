   14 -- 基准测试套件
   15 benchmarkSuite :: Benchmark
   16 benchmarkSuite = bgroup "Compiler Benchmarks" [
   17     bgroup "Parser" [
   18         bench "parse simple code" $ whnfIO $ loadAndParse "simple_go_code.typus",
   19         bench "parse complex code" $ whnfIO $ loadAndParse "complex_ownership_code.typus",
   20         bench "parse dependent types" $ whnfIO $ loadAndParse "code_with_dependent_types.typus",
   21         bench "parse large code" $ whnfIO $ loadAndParse "large_code.typus"
   22     ],
   23     bgroup "Compiler" [
   24         bench "compile simple code" $ whnfIO $ loadAndCompile "simple_go_code.typus",
   25         bench "compile complex code" $ whnfIO $ loadAndCompile "complex_ownership_code.typus",
   26         bench "compile dependent types" $ whnfIO $ loadAndCompile "code_with_dependent_types.typus",
   27         bench "compile large code" $ whnfIO $ loadAndCompile "large_code.typus"

   32         bench "analyze dependent types" $ whnfIO $ loadAndAnalyzeOwnership "code_with_dependent_types.typus",
   33         bench "analyze large code" $ whnfIO $ loadAndAnalyzeOwnership "large_code.typus"
   34     ],
   35     bgroup "End-to-End" [

   42 -- 加载测试数据
   43 loadTestData :: FilePath -> IO String
   44 loadTestData fileName = do
   45     let testDir = "test" </> "data"
   46     let filePath = testDir </> fileName
   47     exists <- doesFileExist filePath
   48     if not exists
   49         then error $ "Test data file not found: " ++ filePath
   50         else readFile filePath
   51 
   52 -- 加载并解析
   53 loadAndParse :: FilePath -> IO ()
   54 loadAndParse fileName = do
   55     code <- loadTestData fileName
   56     _ <- evaluate $ Parser.parseTypus code
   57     return ()
   58 
   59 -- 加载并编译
   60 loadAndCompile :: FilePath -> IO ()
   61 loadAndCompile fileName = do
   62     code <- loadTestData fileName
   63     case Parser.parseTypus code of
   64         Left err -> error $ "Parser failed: " ++ err
   65         Right typusFile -> do
   66             _ <- evaluate $ Compiler.compile typusFile
   67             return ()
   68 
   69 -- 加载并进行所有权分析
   70 loadAndAnalyzeOwnership :: FilePath -> IO ()
   71 loadAndAnalyzeOwnership fileName = do
   72     code <- loadTestData fileName
   73     _ <- evaluate $ Ownership.analyzeOwnership code
   74     return ()
   75 
   76 -- 完整的编译管道
   77 fullPipeline :: FilePath -> IO ()
   78 fullPipeline fileName = do
   79     code <- loadTestData fileName
   80     case Parser.parseTypus code of
   81         Left err -> error $ "Parser failed: " ++ err
   82         Right typusFile -> do
   83             case Compiler.compile typusFile of
   84                 Left err -> error $ "Compilation failed: " ++ err
   85                 Right goCode -> do
   86                     _ <- evaluate $ length goCode
   87                     _ <- evaluate $ "package main" `isInfixOf` goCode
   88                     return ()
   89 
   90 -- 运行基准测试
   91 runBenchmarks :: IO ()
   92 runBenchmarks = do
   93     putStrLn "Running compiler benchmarks..."
   94     putStrLn "================================"
   95     
   96     let config = defaultConfig {
   97         -- 设置更合理的基准测试配置
   98         timeLimit = 30,  -- 每个基准测试最多30秒
   99         resamples = 10,  -- 重采样次数
  100         verbosity = Normal
  101     }
  102     
  103     defaultMain [benchmarkSuite]
