   50 -- 从测试数据目录加载测试用例
   51 loadTestData :: FilePath -> IO String
   52 loadTestData fileName = do
   53     let testDir = "test" </> "data"
   54     let filePath = testDir </> fileName
   55     exists <- doesFileExist filePath
   56     if not exists
   57         then error $ "Test data file not found: " ++ filePath

   60 -- 测试端到端编译流程
   61 testEndToEndCompilation :: TestConfig -> TestTree
   62 testEndToEndCompilation config = testCase "End-to-End Compilation" $ do
   63     setupTestEnvironment config
   64     
   65     code <- loadTestData "simple_go_code.typus"
   66     
   67     -- 解析
   68     case Parser.parseTypus code of
   69         Left err -> assertFailure $ "Parser failed: " ++ err
   70         Right typusFile -> do
   71             -- 编译
   72             case Compiler.compile typusFile of
   73                 Left err -> assertFailure $ "Compilation failed: " ++ err
   74                 Right goCode -> do
   75                     -- 验证生成的 Go 代码
   76                     assertBool "Generated code should contain package declaration" 
   77                         ("package main" `isInfixOf` goCode)
   78                     assertBool "Generated code should contain main function" 
   79                         ("func main" `isInfixOf` goCode)
   80                     assertBool "Generated code should not contain unused crypto imports" 
   81                         (not ("crypto/md5" `isInfixOf` goCode) || "md5." `isInfixOf` goCode)
   82                     assertBool "Generated code should not contain unused encoding imports" 
   83                         (not ("encoding/base64" `isInfixOf` goCode) || "base64." `isInfixOf` goCode)
   84                     
   85                     -- 尝试编译生成的 Go 代码
   86                     let tempGoFile = tempDir config </> "test_output.go"
   87                     writeFile tempGoFile goCode
   88                     
   89                     -- 使用 Go 编译器编译
   90                     (exitCode, _stdout, stderr) <- readProcessWithExitCode 
   91                         (goCompilerPath config) 
   92                         ["build", "-o", tempDir config </> "test_output", tempGoFile] 
   93                         ""
   94                     
   95                     case exitCode of
   96                         ExitSuccess -> do
   97                             -- 尝试运行生成的可执行文件
   98                             (exitCode', _stdout', _stderr') <- readProcessWithExitCode 
   99                                 (tempDir config </> "test_output") 
  100                                 [] 
  101                                 ""
  102                             
  103                             case exitCode' of
  104                                 ExitSuccess -> return ()
  105                                 ExitFailure c -> assertFailure $ "Generated program failed with exit code " ++ show c
  106                         
  107                         ExitFailure c -> do
  108                             if "command not found" `isInfixOf` stderr
  109                                 then do
  110                                     putStrLn "WARNING: Go compiler not found, skipping compilation test"
  111                                     return ()
  112                                 else assertFailure $ "Go compilation failed with exit code " ++ show c ++ ": " ++ stderr

  279 -- 运行集成测试并生成报告
  280 runIntegrationTests :: TestConfig -> IO ()
  281 runIntegrationTests config = do
  282     putStrLn "Running integration test suite for Typus compiler..."
  283     putStrLn "================================================"
  284     putStrLn $ "Test configuration:"
  285     putStrLn $ "  Temporary directory: " ++ tempDir config
  286     putStrLn $ "  Go compiler path: " ++ goCompilerPath config
  287     putStrLn $ "  Timeout: " ++ show (timeoutSeconds config) ++ " seconds"
  288     putStrLn ""
  289     
  290     setupTestEnvironment config
  291     
  292     -- 使用 Tasty 运行集成测试
  293     let ingredients = [consoleTestReporter]
  294     case tryIngredients ingredients mempty (integrationTestSuite config) of
  295         Nothing -> do
  296             putStrLn "ERROR: No suitable ingredient found to run integration tests"
  297         Just runTest -> do
  298             success <- runTest
  299             if success
  300                 then do
  301                     cleanupTestEnvironment config
  302                     putStrLn "SUCCESS: All integration tests passed. The project is ready for production."
  303                 else do
  304                     putStrLn "ERROR: Some integration tests failed."
  305     return ()
