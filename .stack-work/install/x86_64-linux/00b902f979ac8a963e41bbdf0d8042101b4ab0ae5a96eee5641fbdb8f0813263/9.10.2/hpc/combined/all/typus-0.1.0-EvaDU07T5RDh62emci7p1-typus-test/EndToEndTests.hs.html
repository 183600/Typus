   38 -- Setup and cleanup
   39 setupE2ETestEnvironment :: E2ETestConfig -> IO ()
   40 setupE2ETestEnvironment config = do
   41     createDirectoryIfMissing True (tempDir config)
   42 
   43 cleanupE2ETestEnvironment :: E2ETestConfig -> IO ()
   44 cleanupE2ETestEnvironment config = do
   45     exists <- doesFileExist (tempDir config)
   46     when exists $ removeDirectoryRecursive (tempDir config)

   48 -- Helper function to test full compilation and execution pipeline
   49 testCompilationAndExecution :: E2ETestConfig -> String -> String -> TestTree
   50 testCompilationAndExecution config testName code = testCase testName $ do
   51     setupE2ETestEnvironment config
   52 
   53     -- Step 1: Parse the code
   54     parsedFile <- case Parser.parseTypus code of
   55         Left err -> assertFailure $ "Parsing failed: " ++ err
   56         Right typusFile -> return typusFile
   57 
   58     -- Step 2: Compile to Go
   59     goCode <- case Compiler.compile parsedFile of
   60         Left err -> assertFailure $ "Compilation failed: " ++ err
   61         Right compiledCode -> return compiledCode
   62 
   63     -- Step 3: Validate generated Go code
   64     assertBool (testName ++ ": Generated Go code should contain package declaration")
   65         ("package main" `isInfixOf` goCode)
   66     assertBool (testName ++ ": Generated Go code should contain main function")
   67         ("func main" `isInfixOf` goCode)
   68 
   69     -- Step 4: Write Go code to temporary file
   70     let goFile = tempDir config </> testName ++ ".go"
   71     writeFile goFile goCode
   72 
   73     -- Step 5: Compile Go code with timeout
   74     let executableFile = tempDir config </> testName
   75     compileResult <- timeout (timeoutSeconds config * 1000000) $
   76         readProcessWithExitCode (goCompilerPath config) ["build", "-o", executableFile, goFile] ""
   77 
   78     case compileResult of
   79         Nothing -> assertFailure (testName ++ ": Go compilation timed out")
   80         Just (exitCode, stdout, stderr) -> case exitCode of
   81             ExitFailure c -> do
   82                 -- Check if go compiler is available
   83                 if "command not found" `isInfixOf` stderr || "not found" `isInfixOf` stderr
   84                     then do
   85                         putStrLn $ "WARNING: Go compiler not found, skipping execution test for " ++ testName
   86                         cleanupE2ETestEnvironment config
   87                     else assertFailure (testName ++ ": Go compilation failed with exit code " ++ show c ++ ": " ++ stderr)
   88             ExitSuccess -> do
   89                 -- Step 6: Execute the compiled program with timeout
   90                 execResult <- timeout (timeoutSeconds config * 1000000) $
   91                     readProcessWithExitCode executableFile [] ""
   92 
   93                 case execResult of
   94                     Nothing -> assertFailure (testName ++ ": Program execution timed out")
   95                     Just (exitCode', stdout', stderr') -> case exitCode' of
   96                         ExitFailure c' -> assertFailure (testName ++ ": Program execution failed with exit code " ++ show c' ++ ": " ++ stderr')
   97                         ExitSuccess -> do
   98                             -- Step 7: Validate output (if applicable)
   99                             when (not $ null stdout') $ do
  100                                 -- Basic output validation
  101                                 assertBool (testName ++ ": Program should produce output") (not $ null stdout')
  102 
  103                             cleanupE2ETestEnvironment config
  104                             return ()

  288 -- End-to-End test runner
  289 runEndToEndTests :: E2ETestConfig -> IO ()
  290 runEndToEndTests config = do
  291     putStrLn "Running end-to-end test suite for Typus compiler..."
  292     putStrLn "================================================"
  293     putStrLn $ "Test configuration:"
  294     putStrLn $ "  Temporary directory: " ++ tempDir config
  295     putStrLn $ "  Go compiler path: " ++ goCompilerPath config
  296     putStrLn $ "  Timeout: " ++ show (timeoutSeconds config) ++ " seconds"
  297     putStrLn ""
  298 
  299     setupE2ETestEnvironment config
  300 
  301     -- Run the test suite
  302     let ingredients = [consoleTestReporter]
  303     case tryIngredients ingredients mempty (endToEndTests config) of
  304         Nothing -> putStrLn "ERROR: No suitable ingredient found to run end-to-end tests"
  305         Just runTest -> do
  306             success <- runTest
  307             if success
  308                 then do
  309                     cleanupE2ETestEnvironment config
  310                     putStrLn "SUCCESS: All end-to-end tests passed. The project is ready for production."
  311                 else do
  312                     cleanupE2ETestEnvironment config
  313                     putStrLn "ERROR: Some end-to-end tests failed."
  314     return ()
