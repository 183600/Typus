   32 -- 从测试数据目录加载测试用例
   33 loadTestData :: FilePath -> IO String
   34 loadTestData fileName = do
   35     let testDir = "test" </> "data"
   36     let filePath = testDir </> fileName
   37     exists <- doesFileExist filePath
   38     when (not exists) $ error $ "Test data file not found: " ++ filePath
   39     readFile filePath

   44 testEndToEndCompilation = testCase "End-to-End Compilation" $ do
   45     code <- loadTestData "simple_go_code.typus"

   64         Left err -> assertFailure $ "Parser failed on large code: " ++ err
   65         Right typusFile -> do
   66             parseEnd <- getCurrentTime
   67             let parseTime = diffUTCTime parseEnd parseStart
   68             assertBool "Parsing should complete within reasonable time"

   71             -- 测试编译性能

   82 testEdgeCases :: TestTree
   83 testEdgeCases = testCase "Edge Cases" $ do
   84     -- 测试空代码
   85     case Parser.parseTypus "" of
   86         Left err -> assertFailure $ "Parser failed on empty code: " ++ err
   87         Right typusFile -> do
   88             assertEqual "Empty code should have no blocks" [] (Parser.tfBlocks typusFile)
   89     
   90     -- 测试只有指令的代码
   91     let directiveOnly = unlines ["//! ownership: on", "//! dependent_types: on"]
   92     case Parser.parseTypus directiveOnly of
   93         Left err -> assertFailure $ "Parser failed on directive-only code: " ++ err
   94         Right typusFile -> do
   95             let directives = Parser.tfDirectives typusFile
   96             assertEqual "Ownership should be enabled" (Just True) (Parser.fdOwnership directives)
   97             assertEqual "Dependent types should be enabled" (Just True) (Parser.fdDependentTypes directives)

  103     malformedCode <- loadTestData "malformed_syntax_code.typus"

  105         Left err -> assertFailure $ "Parser should not fail on malformed syntax (it only parses directives): " ++ err
  106         Right typusFile -> do
  107             -- 解析器应该成功，因为它只解析指令，不验证代码语法
  108             assertEqual "Malformed code should be parsed as a code block" 1 (length $ Parser.tfBlocks typusFile)
  109

  111     ownershipErrorCode <- loadTestData "use_after_move_code.typus"

  141 -- 属性测试：解析器往返测试
  142 parserRoundtrip :: String -> Bool
  143 parserRoundtrip code = 
  144     let parsed = Parser.parseTypus code
  145     in case parsed of
  146         Left _ -> True  -- 无效代码是允许的
  147         Right _ -> True  -- 简化版本，实际项目中应该有更完整的往返测试

  149 -- 属性测试：编译器输出有效性
  150 compilerOutputValidity :: String -> Bool
  151 compilerOutputValidity code = 
  152     case Parser.parseTypus code of
  153         Left _ -> True
  154         Right typusFile -> 
  155             case Compiler.compile typusFile of
  156                 Left _ -> True
  157                 Right goCode -> 
  158                     -- 检查生成的Go代码是否包含基本结构
  159                     "package main" `isInfixOf` goCode || 
  160                     (null goCode && null code)  -- 空输入产生空输出
