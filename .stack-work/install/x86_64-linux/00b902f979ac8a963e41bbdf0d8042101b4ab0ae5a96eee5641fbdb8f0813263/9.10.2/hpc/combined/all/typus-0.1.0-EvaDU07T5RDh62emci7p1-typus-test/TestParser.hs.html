    7 -- Test cases for Parser module
    8 testParserBasicFunctionality :: Test
    9 testParserBasicFunctionality = TestCase $ do
   10     let code = unlines [
   11             "package main",
   12             "",
   13             "func main() {",
   14             "    fmt.Println(\"Hello, World!\")",
   15             "}"
   16             ]
   17     let result = Parser.parseTypus code
   18     case result of
   19         Left err -> assertFailure $ "Parser failed on simple Go code: " ++ err
   20         Right typusFile -> do
   21             assertBool "Should parse package directive correctly" 
   22                 (Parser.tfDirectives typusFile == Parser.FileDirectives Nothing Nothing Nothing)
   23             assertBool "Should have at least one code block" 
   24                 (not $ null $ Parser.tfBlocks typusFile)
   25 
   26 testParserFileDirectives :: Test
   27 testParserFileDirectives = TestCase $ do
   28     let code = unlines [
   29             "//! ownership: on",
   30             "//! dependent_types: on",
   31             "",
   32             "package main",
   33             "",
   34             "func main() {",
   35             "    fmt.Println(\"Hello, World!\")",
   36             "}"
   37             ]
   38     let result = Parser.parseTypus code
   39     case result of
   40         Left err -> assertFailure $ "Parser failed on code with directives: " ++ err
   41         Right typusFile -> do
   42             let directives = Parser.tfDirectives typusFile
   43             assertEqual "Ownership should be enabled" (Just True) (Parser.fdOwnership directives)
   44             assertEqual "Dependent types should be enabled" (Just True) (Parser.fdDependentTypes directives)
   45 
   46 testParserEmptyCode :: Test
   47 testParserEmptyCode = TestCase $ do
   48     let result = Parser.parseTypus ""
   49     case result of
   50         Left err -> assertFailure $ "Parser failed on empty code: " ++ err
   51         Right typusFile -> do
