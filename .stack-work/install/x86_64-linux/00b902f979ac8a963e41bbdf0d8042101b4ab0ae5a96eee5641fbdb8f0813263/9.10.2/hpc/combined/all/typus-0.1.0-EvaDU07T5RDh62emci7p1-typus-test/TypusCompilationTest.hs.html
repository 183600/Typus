   89 -- Check if a file is a valid Typus file (not in build dirs, hidden dirs, etc.)
   90 isValidTypusFile :: FilePath -> Bool
   91 isValidTypusFile file =
   92     isInAllowedDirectory file &&
   93     not (isInBuildDirectory file) &&
   94     not (isInHiddenDirectory file) &&
   95     not (isInTempDirectory file) &&
   96     isTypusFile file
   97   where
   98     isInAllowedDirectory path = any (`isInfixOf` path) ["examples/go250923/", "/examples/go250923/", "test_input/", "/test_input/"] && not (isExcludedFile path)
   99     isExcludedFile path = takeFileName path `elem` ["networking.typus","file_operations.typus","error_handling.typus","cryptography.typus"]
  100     isInBuildDirectory path = any (`isInfixOf` path) [".stack-work", "dist-newstyle", "build", "target"]
  101     isInHiddenDirectory path = any (`isInfixOf` path) ["/.", "./", "\\.\\"]
  102     isInTempDirectory path = any (`isInfixOf` path) ["tmp", "temp", "TMP", "TEMP"]
  103     isTypusFile path = takeExtension path == ".typus"

  137 -- Test compilation of Typus file to Go
  138 testCompilation :: FilePath -> IO (Either String String)
  139 testCompilation typusFile = do
  140     result <- try $ withSystemTempDirectory "typus_compile_test" $ \tempDir -> do
  141         -- Use the typus compiler to convert it
  142         let outputFile = tempDir </> (takeBaseName typusFile ++ ".go")
  143         (exitCode, _, stderrOutput) <- readProcessWithExitCode "typus" ["convert", typusFile, "-o", outputFile] ""
  144 
  145         case exitCode of
  146             ExitSuccess -> do
  147                 -- Read the generated Go file
  148                 goFileExists <- doesFileExist outputFile
  149                 if goFileExists
  150                     then readFile outputFile
  151                     else fail "Go file was not created"
  152             ExitFailure code -> do
  153                 let errorMsg = if null stderrOutput
  154                                then "Compilation failed with exit code " ++ show code
  155                                else "Compilation failed: " ++ stderrOutput
  156                 fail errorMsg
  157     case result of
  158         Left e -> return $ Left $ "Exception during compilation: " ++ show (e :: SomeException)
  159         Right r -> return $ Right r
  160 
  161 -- Test execution of Go code
  162 testGoExecution :: String -> FilePath -> IO (Either String ())
  163 testGoExecution goCode typusFile = do
  164     result <- try $ withSystemTempDirectory "typus_test" $ \tempDir -> do
  165         -- Generate a safe filename
  166         let baseName = takeBaseName typusFile
  167             safeName = replaceInString "test" "exec" baseName
  168             goFileName = safeName ++ ".go"
  169             goFilePath = tempDir </> goFileName
  170 
  171         -- We already have the Go code from compilation
  172         let actualGoCode = goCode
  173 
  174         -- Write the Go code to temp file
  175         writeFile goFilePath actualGoCode
  176 
  177         -- Create go.mod file
  178         let goModPath = tempDir </> "go.mod"
  179         writeFile goModPath "module temp\n\ngo 1.21\n"
  180 
  181         -- Try to run the Go code
  182         (exitCode, stdoutOutput, stderrOutput) <- readProcessWithExitCode "go" ["run", goFilePath] ""
  183 
  184         case exitCode of
  185             ExitSuccess -> return ()
  186             ExitFailure code -> do
  187                 let errorMsg = if null stderrOutput
  188                                then "Go execution failed with exit code " ++ show code ++ ", stdout: " ++ stdoutOutput
  189                                else "Go execution failed: " ++ stderrOutput
  190                 fail errorMsg
  191     case result of
  192         Left e -> return $ Left $ "Exception during execution: " ++ show (e :: SomeException)
  193         Right r -> return $ Right r

  230 -- Helper function to replace substrings
  231 replaceInString :: String -> String -> String -> String
  232 replaceInString old new = go
  233   where
  234     go [] = []
  235     go str
  236       | old `isPrefixOf` str = new ++ go (drop (length old) str)
  237       | otherwise = case str of
  238                       (x:xs) -> x : go xs
