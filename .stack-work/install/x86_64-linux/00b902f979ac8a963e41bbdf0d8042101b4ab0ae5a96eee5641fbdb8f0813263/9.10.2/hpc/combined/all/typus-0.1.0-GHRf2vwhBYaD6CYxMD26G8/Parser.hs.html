   46 data Token

   53     | TWhitespace

   77 {-
   78 -- Get current character
   79 currentChar :: Tokenizer (Maybe Char)
   80 currentChar = do
   81     input <- gets tsInput
   82     case input of
   83         [] -> return Nothing
   84         (c:_) -> return (Just c)

   87 {-
   88 -- Advance to next character
   89 advance :: Tokenizer ()
   90 advance = do
   91     st <- get
   92     case tsInput st of
   93         [] -> return ()
   94         (c:rest) -> put st
   95             { tsInput = rest
   96             , tsPosition = tsPosition st + 1
   97             , tsColumn = if c == '\n' then 1 else tsColumn st + 1
   98             , tsLine = if c == '\n' then tsLine st + 1 else tsLine st
   99             }
  100 -}

  103 -- Tokenizer error
  104 tokenizerError :: String -> Tokenizer a
  105 tokenizerError msg = do
  106     pos <- gets tsPosition
  107     lift $ Left $ "Tokenizer error at position " ++ show pos ++ ": " ++ msg
  108 -}

  110 {-
  111 -- Skip whitespace

  113 skipWhitespace = do
  114     mc <- currentChar

  116         Just c | isSpace c -> advance >> skipWhitespace

  170 {-
  171 -- Parse string content
  172 parseStringContent :: Tokenizer String
  173 parseStringContent = do
  174     mc <- currentChar
  175     case mc of
  176         Just '"' -> do
  177             advance  -- Skip closing quote
  178             return ""
  179         Just '\\' -> do
  180             advance  -- Skip escape character
  181             mc' <- currentChar
  182             case mc' of
  183                 Just c -> do
  184                     advance
  185                     rest <- parseStringContent
  186                     return (c:rest)
  187                 Nothing -> tokenizerError "Unexpected end of input in string literal"
  188         Just c -> do
  189             advance
  190             rest <- parseStringContent
  191             return (c:rest)
  192         Nothing -> tokenizerError "Unexpected end of input in string literal"
  193 -}

  256 {-
  257 -- Parse directive (starts with //!)
  258 parseDirective :: Tokenizer Token
  259 parseDirective = do
  260     input <- gets tsInput
  261     if "//!" `isPrefixOf` input
  262         then do
  263             replicateM_ 3 advance
  264             content <- parseLineContent
  265             return $ TDirective content
  266         else tokenizerError "Expected directive marker //!"
  267 -}

  269 {-
  270 -- Parse comment (starts with // but not //!)
  271 parseComment :: Tokenizer Token
  272 parseComment = do
  273     input <- gets tsInput
  274     if "//" `isPrefixOf` input && not ("//!" `isPrefixOf` input)
  275         then do
  276             replicateM_ 2 advance
  277             content <- parseLineContent
  278             return $ TComment content
  279         else tokenizerError "Expected comment marker //"
  280 -}

  282 {-
  283 -- Helper to parse content until end of line
  284 parseLineContent :: Tokenizer String
  285 parseLineContent = do
  286     mc <- currentChar
  287     case mc of
  288         Just '\n' -> return ""  -- End of line
  289         Just c -> do
  290             advance
  291             rest <- parseLineContent
  292             return (c:rest)
  293         Nothing -> return ""  -- End of input
  294 -}

  296 {-
  297 -- Main tokenization function (fixed order for / vs symbol; doesn't emit TWhitespace)
  298 tokenize :: String -> Either String [Token]
  299 tokenize input = evalStateT (tokenize') (initTokenizer input)
  300   where
  301     tokenize' :: Tokenizer [Token]
  302     tokenize' = do
  303         skipWhitespace
  304         mc <- currentChar
  305         case mc of
  306             Nothing -> return [TEOF]
  307             Just '/' -> do
  308                 input' <- gets tsInput
  309                 tok <- if "//!" `isPrefixOf` input'
  310                       then parseDirective
  311                       else if "//" `isPrefixOf` input'
  312                            then parseComment
  313                            else parseSymbol
  314                 rest <- tokenize'
  315                 return (tok:rest)
  316             Just '"' -> do
  317                 tok <- parseStringLiteral
  318                 rest <- tokenize'
  319                 return (tok:rest)
  320             Just c
  321                 | isDigit c -> do
  322                     tok <- parseNumberLiteral
  323                     rest <- tokenize'
  324                     return (tok:rest)
  325                 | isAlpha c || c == '_' -> do
  326                     tok <- parseIdentifier
  327                     rest <- tokenize'
  328                     return (tok:rest)
  329                 | isSymbolChar c -> do
  330                     tok <- parseSymbol
  331                     rest <- tokenize'
  332                     return (tok:rest)
  333                 | isSpace c -> do
  334                     -- skipWhitespace already consumed them; just continue
  335                     advance
  336                     tokenize'
  337                 | otherwise -> tokenizerError ("Unexpected character: " ++ [c])
  338 -}
  339 -}

  347 -- High-level parser that works line-by-line to handle directives and blocks.
  348 parseTypus :: String -> Either String TypusFile
  349 parseTypus input = do
  350     let ls = map stripCR (lines input)
  351     (fileDirs, buildTags, rest) <- parseFileDirectivesFromLines ls
  352     blocks <- parseBlocksFromLines rest
  353     return $ TypusFile fileDirs buildTags blocks
  354 
  355 -- Parse file-level directives at the very top of file.
  356 -- It skips leading blank lines, and accepts consecutive "//! key: value" lines.
  357 -- Also preserves Go build tags (//go:build and // +build) at the very top.
  358 -- Stops at the first non-blank non-file-directive non-build-tag line.
  359 parseFileDirectivesFromLines :: [String] -> Either String (FileDirectives, [String], [String])
  360 parseFileDirectivesFromLines = go defaultFileDirectives []
  361   where
  362     go :: FileDirectives -> [String] -> [String] -> Either String (FileDirectives, [String], [String])
  363     go acc buildTags [] = Right (acc, buildTags, [])
  364     go acc buildTags (l:ls) =
  365         let t = trim l
  366         in if t == ""
  367             then go acc buildTags ls
  368             else if isPrefixOf "//!" t
  369                 then do
  370                     (k, v) <- parseFileDirectiveLine t
  371                     acc' <- updateFileDirective acc k v
  372                     go acc' buildTags ls
  373             else if isBuildTagLine t
  374                 then -- Preserve build tags by keeping them in the buildTags list
  375                      go acc (buildTags ++ [l]) ls
  376                 else Right (acc, buildTags, l:ls)
  377     
  378     isBuildTagLine line = 
  379         isPrefixOf "//go:build" line || isPrefixOf "// +build" line
  380 
  381 -- Parse blocks:
  382 -- - Normal code is accumulated as a default block (if non-empty).
  383 -- - "{//! ...}" starts a directive block; we gather lines until the "extra }"
  384 --   that closes the directive block. Any inner code braces are balanced and
  385 --   do not end the directive block prematurely.
  386 parseBlocksFromLines :: [String] -> Either String [CodeBlock]
  387 parseBlocksFromLines = go [] []
  388   where
  389     -- go accBlocks accCodeBuf remainingLines
  390     go :: [CodeBlock] -> [String] -> [String] -> Either String [CodeBlock]
  391     go acc blocksBuf [] =
  392         let finalBlocks = acc ++ flushCodeBuf blocksBuf
  393         in Right finalBlocks
  394     go acc blocksBuf (l:ls) =
  395         let t = trim l
  396         in if startsWithBlockDirective t
  397             then do
  398                 -- Flush any pending normal code into a default block
  399                 let acc' = acc ++ flushCodeBuf blocksBuf
  400                 -- Parse block directives
  401                 kvs <- parseBlockDirectiveLine t
  402                 bd <- parseBlockDirectives kvs
  403                 -- Capture block content until the directive-closing '}'
  404                 (blockLines, rest) <- captureDirectiveBlock ls
  405                 let codeTxt = unlines blockLines
  406                 let blk = CodeBlock bd codeTxt
  407                 go (acc' ++ [blk]) [] rest
  408             else
  409                 -- Accumulate normal code
  410                 go acc (blocksBuf ++ [l]) ls
  411 
  412     flushCodeBuf :: [String] -> [CodeBlock]
  413     flushCodeBuf buf =
  414         let codeTxt = trimRight (unlines buf)
  415         in if codeTxt == "" then [] else [CodeBlock defaultBlockDirectives codeTxt]
  416 
  417     startsWithBlockDirective :: String -> Bool
  418     startsWithBlockDirective s = isPrefixOf "{//!" (dropWhile isSpace s)
  419 
  420 -- Capture lines of a directive block until we see the "extra }" that closes it.
  421 -- We keep a brace-depth over code to avoid stopping at function's }.
  422 -- The closing '}' for the directive block will make the depth go from 0 to -1.
  423 captureDirectiveBlock :: [String] -> Either String ([String], [String])
  424 captureDirectiveBlock = go 0 []
  425   where
  426     go :: Int -> [String] -> [String] -> Either String ([String], [String])
  427     go _ _acc [] = Left "Unclosed directive block: missing closing '}'"
  428     go depth acc (l:ls) =
  429         let d = curlyDelta l
  430             newDepth = depth + d
  431         in if newDepth < 0
  432             then -- The last line l is the directive-closing '}', do not include it
  433                  Right (acc, ls)
  434             else go newDepth (acc ++ [l]) ls
  435 
  436 -- Compute net curly-brace delta for a line, ignoring braces inside strings and line-comments.
  437 -- Strings: "..." with support for escaping \" inside (approx).
  438 -- Line comments: // ... (ignored).
  439 curlyDelta :: String -> Int
  440 curlyDelta = go False False 0 0
  441   where
  442     go :: Bool -> Bool -> Int -> Int -> String -> Int
  443     go _ _ acc _ [] = acc
  444     go inStr _esc acc _ ('/':'/':_) | not inStr = acc  -- comment starts, ignore rest
  445     go inStr _esc acc _prev (c:cs)
  446         | inStr =
  447             case c of
  448                 '"' | not _esc -> go False False acc (fromEnum c) cs
  449                 '\\' -> go True True acc (fromEnum c) cs
  450                 _    -> go True False acc (fromEnum c) cs
  451         | otherwise =
  452             case c of
  453                 '"'  -> go True False acc (fromEnum c) cs
  454                 '{'  -> go False False (acc + 1) (fromEnum c) cs
  455                 '}'  -> go False False (acc - 1) (fromEnum c) cs
  456                 _    -> go False False acc (fromEnum c) cs

  462 parseFileDirectiveLine :: String -> Either String (String, String)
  463 parseFileDirectiveLine line =
  464     let directivePart = trim (drop 3 (dropWhile isSpace line))  -- Drop "//!"
  465     in case break (== ':') directivePart of
  466         (key, ':':value) -> Right (trim key, trim value)
  467         _ -> Left $ "Invalid file directive format: " ++ line
  468 
  469 parseBlockDirectiveLine :: String -> Either String [(String, String)]
  470 parseBlockDirectiveLine line = do
  471     -- Expect a line like: {//! ownership: off, constraints: on}
  472     let tline = dropWhile isSpace line
  473     if not ("{//!" `isPrefixOf` tline)
  474        then Left $ "Invalid block directive line (missing {//!): " ++ line
  475        else do
  476            -- Extract content between {//! and the matching }
  477            let afterPrefix = trim (drop 4 tline)  -- drop "{//!"
  478            let content = takeWhile (/= '}') afterPrefix
  479            if null (trim content)
  480            then Right []
  481            else mapM parseKeyValue $ splitOn ',' content
  482   where
  483     parseKeyValue :: String -> Either String (String, String)
  484     parseKeyValue s = case break (== ':') (trim s) of
  485         (key, ':':value) -> Right (trim key, trim value)
  486         _ -> Left $ "Invalid key:value format: " ++ s
  487 
  488 updateFileDirective :: FileDirectives -> String -> String -> Either String FileDirectives
  489 updateFileDirective fd key value = do
  490     boolValue <- parseBool value
  491     case key of
  492         "ownership" -> Right fd { fdOwnership = Just boolValue }
  493         "dependent_types" -> Right fd { fdDependentTypes = Just boolValue }
  494         "constraints" -> Right fd { fdConstraints = Just boolValue }
  495         _ -> Left $ "Unknown file directive: " ++ key
  496 
  497 parseBlockDirectives :: [(String, String)] -> Either String BlockDirectives
  498 parseBlockDirectives [] = Right defaultBlockDirectives
  499 parseBlockDirectives ((key, value):kvs) = do
  500     bd <- updateBlockDirective defaultBlockDirectives (key, value)
  501     parseBlockDirectives' bd kvs
  502 
  503 parseBlockDirectives' :: BlockDirectives -> [(String, String)] -> Either String BlockDirectives
  504 parseBlockDirectives' bd [] = Right bd
  505 parseBlockDirectives' bd ((key, value):kvs) = do
  506     bd' <- updateBlockDirective bd (key, value)
  507     parseBlockDirectives' bd' kvs
  508 
  509 updateBlockDirective :: BlockDirectives -> (String, String) -> Either String BlockDirectives
  510 updateBlockDirective bd (key, value) = do
  511     boolValue <- parseBool value
  512     case key of
  513         "ownership" -> Right bd { bdOwnership = boolValue }
  514         "dependent_types" -> Right bd { bdDependentTypes = boolValue }
  515         "constraints" -> Right bd { bdConstraints = boolValue }
  516         _ -> Left $ "Unknown block directive: " ++ key
  517 
  518 parseBool :: String -> Either String Bool
  519 parseBool s = case trim s of
  520     "on" -> Right True
  521     "off" -> Right False
  522     "true" -> Right True
  523     "false" -> Right False
  524     v -> Left $ "Invalid boolean value: " ++ v

  530 trim :: String -> String

  534 trimRight :: String -> String

  537 {-
  538 breakOn :: String -> String -> (String, String)
  539 breakOn delimiter str = go str
  540   where
  541     go [] = ("", "")
  542     go s@(x:xs)
  543         | delimiter `isPrefixOf` s = ("", s)
  544         | otherwise = let (before, after) = go xs in (x:before, after)
  545 -}

  547 splitOn :: Char -> String -> [String]
  548 splitOn _ [] = []
  549 splitOn delim str = case break (== delim) str of
  550     (before, []) -> [before]
  551     (before, _:after) -> before : splitOn delim after
  552 
  553 isPrefixOf :: String -> String -> Bool
  554 isPrefixOf [] _ = True
  555 isPrefixOf _ [] = False
  556 isPrefixOf (x:xs) (y:ys) = x == y && isPrefixOf xs ys
