   16 -- Comprehensive unit tests for Parser module
   17 parserComprehensiveTests :: TestTree
   18 parserComprehensiveTests = testGroup "Parser Comprehensive Tests" [
   19     testCase "Parser handles empty input" $ do
   20         result <- evaluate $ Parser.parseTypus ""
   21         case result of
   22             Left err -> assertFailure $ "Parser failed on empty input: " ++ err
   23             Right typusFile -> do
   24                 assertEqual "Empty input should have no blocks" [] (Parser.tfBlocks typusFile)
   25                 assertEqual "Empty input should have default directives" Parser.defaultFileDirectives (Parser.tfDirectives typusFile),
   26 
   27     testCase "Parser handles single directive" $ do
   28         let code = "//! ownership: on"
   29         result <- evaluate $ Parser.parseTypus code
   30         case result of
   31             Left err -> assertFailure $ "Parser failed on single directive: " ++ err
   32             Right typusFile -> do
   33                 let directives = Parser.tfDirectives typusFile
   34                 assertEqual "Ownership should be enabled" (Just True) (Parser.fdOwnership directives)
   35                 assertEqual "Dependent types should be Nothing" Nothing (Parser.fdDependentTypes directives),
   36 
   37     testCase "Parser handles multiple directives" $ do
   38         let code = unlines ["//! ownership: on", "//! dependent_types: on", "//! constraints: on"]
   39         result <- evaluate $ Parser.parseTypus code
   40         case result of
   41             Left err -> assertFailure $ "Parser failed on multiple directives: " ++ err
   42             Right typusFile -> do
   43                 let directives = Parser.tfDirectives typusFile
   44                 assertEqual "Ownership should be enabled" (Just True) (Parser.fdOwnership directives)
   45                 assertEqual "Dependent types should be enabled" (Just True) (Parser.fdDependentTypes directives)
   46                 assertEqual "Constraints should be enabled" (Just True) (Parser.fdConstraints directives),
   47 
   48     testCase "Parser handles mixed content" $ do
   49         let code = unlines [
   50                 "//! ownership: on",
   51                 "package main",
   52                 "",
   53                 "func main() {",
   54                 "    println(\"Hello, World!\")",
   55                 "}"
   56                 ]
   57         result <- evaluate $ Parser.parseTypus code
   58         case result of
   59             Left err -> assertFailure $ "Parser failed on mixed content: " ++ err
   60             Right typusFile -> do
   61                 let directives = Parser.tfDirectives typusFile
   62                 let blocks = Parser.tfBlocks typusFile
   63                 assertEqual "Ownership should be enabled" (Just True) (Parser.fdOwnership directives)
   64                 assertBool "Should have at least one code block" (not $ null blocks),
   65 
   66     testCase "Parser handles malformed directives gracefully" $ do
   67         let code = "//! invalid_directive: value"
   68         result <- evaluate $ Parser.parseTypus code
   69         case result of
   70             Left _ -> return ()  -- Expected to fail
   71             Right typusFile -> do
   72                 -- Should still parse with default directives
   73                 let directives = Parser.tfDirectives typusFile
   74                 assertEqual "Should have default directives for invalid directive" Parser.defaultFileDirectives directives,
   75 
   76     testCase "Parser handles Unicode content" $ do
   77         let code = unlines ["//! ownership: on", "func main() { println(\"你好，世界！\") }"]
   78         result <- evaluate $ Parser.parseTypus code
   79         case result of
   80             Left err -> assertFailure $ "Parser failed on Unicode content: " ++ err
   81             Right typusFile -> do
   82                 assertBool "Should parse Unicode content successfully" (not $ null $ Parser.tfBlocks typusFile),
   83 
   84     testProperty "Parser is deterministic" $ \code ->
   85         let result1 = Parser.parseTypus code
   86             result2 = Parser.parseTypus code
   87         in case (result1, result2) of
   88             (Left _, Left _) -> True
   89             (Right ast1, Right ast2) -> ast1 == ast2
   90             _ -> False
   91     ]

  179                 assertBool "Generated code should contain func main" ("func main" `isInfixOf` goCode)
  180                 assertBool "Generated code should contain proper braces" ("{" `isInfixOf` goCode && "}" `isInfixOf` goCode),
