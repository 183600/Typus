   37 -- 从测试数据目录加载测试用例
   38 loadTestData :: FilePath -> IO String
   39 loadTestData fileName = do
   40     let testDir = "test" </> "data"
   41     let filePath = testDir </> fileName
   42     exists <- doesFileExist filePath
   43     if not exists
   44         then error $ "Test data file not found: " ++ filePath
   45         else readFile filePath
   46 
   47 -- 测试解析性能
   48 testParsePerformance :: PerformanceConfig -> TestTree
   49 testParsePerformance config = testCase "Parse Performance" $ do
   50     code <- loadTestData "large_code.typus"
   51     
   52     start <- getCurrentTime
   53     result <- evaluate $ Parser.parseTypus code
   54     end <- getCurrentTime
   55     
   56     let parseTime :: Double
   57         parseTime = realToFrac $ diffUTCTime end start
   58     
   59     case result of
   60         Left err -> assertFailure $ "Parser failed: " ++ err
   61         Right _ -> do
   62             assertBool ("Parsing should complete within " ++ show (maxParseTime config) ++ " seconds")
   63                 (parseTime <= maxParseTime config)
   64 
   65 -- 测试编译性能
   66 testCompilePerformance :: PerformanceConfig -> TestTree
   67 testCompilePerformance config = testCase "Compile Performance" $ do
   68     code <- loadTestData "large_code.typus"
   69     
   70     case Parser.parseTypus code of
   71         Left err -> assertFailure $ "Parser failed: " ++ err
   72         Right typusFile -> do
   73             start <- getCurrentTime
   74             result <- evaluate $ Compiler.compile typusFile
   75             end <- getCurrentTime
   76             
   77             let compileTime :: Double
   78                 compileTime = realToFrac $ diffUTCTime end start
   79             
   80             case result of
   81                 Left err -> assertFailure $ "Compilation failed: " ++ err
   82                 Right _ -> do
   83                     assertBool ("Compilation should complete within " ++ show (maxCompileTime config) ++ " seconds")
   84                         (compileTime <= maxCompileTime config)
   85

  116 -- 测试并发性能
  117 testConcurrentPerformance :: PerformanceConfig -> TestTree
  118 testConcurrentPerformance config = testCase "Concurrent Performance" $ do
  119     -- 测试并发处理多个文件的能力
  120     let testFiles :: [FilePath]
  121         testFiles = ["simple_go_code.typus", "complex_ownership_code.typus", "code_with_dependent_types.typus"]
  122     
  123     start <- getCurrentTime
  124     results <- mapM loadTestData testFiles
  125     parseResults <- mapM (evaluate . Parser.parseTypus) results
  126     
  127     -- 检查所有解析结果
  128     mapM_ (\case
  129         Left err -> assertFailure $ "Parser failed: " ++ err
  130         Right _ -> return ()) parseResults
  131     
  132     end <- getCurrentTime
  133     let concurrentTime :: Double
  134         concurrentTime = realToFrac $ diffUTCTime end start
  135     
  136     assertBool ("Concurrent parsing should complete within " ++ show (maxParseTime config) ++ " seconds")
  137         (concurrentTime <= maxParseTime config)

  149 -- 运行性能测试并生成报告
  150 runPerformanceTests :: PerformanceConfig -> IO ()
  151 runPerformanceTests config = do
  152     putStrLn "Running performance test suite for Typus compiler..."
  153     putStrLn "================================================"
  154     putStrLn $ "Performance thresholds:"
  155     putStrLn $ "  Max parse time: " ++ show (maxParseTime config) ++ " seconds"
  156     putStrLn $ "  Max compile time: " ++ show (maxCompileTime config) ++ " seconds"
  157     putStrLn $ "  Max ownership time: " ++ show (maxOwnershipTime config) ++ " seconds"
  158     putStrLn $ "  Max memory usage: " ++ show (maxMemoryUsage config) ++ " MB"
  159     putStrLn ""
  160     
  161     -- 使用 Tasty 运行性能测试
  162     let ingredients = [consoleTestReporter]
  163     case tryIngredients ingredients mempty (performanceTestSuite config) of
  164         Nothing -> do
  165             putStrLn "ERROR: No suitable ingredient found to run performance tests"
  166         Just runTest -> do
  167             success <- runTest
  168             if success
  169                 then putStrLn "SUCCESS: All performance tests passed. The project meets production requirements."
  170                 else putStrLn "ERROR: Some performance tests failed."
  171     return ()
