   65 -- Test that typus convert command
   66 testTypusConvert :: FilePath -> FilePath -> IO Bool
   67 testTypusConvert inputDir outputDir = do
   68     -- Clean up previous output
   69     outputExists <- doesDirectoryExist outputDir
   70     when outputExists $ removeDirectoryRecursive outputDir
   71 
   72     -- Run typus convert command
   73     putStrLn $ "Running: stack exec -- typus convert " ++ inputDir ++ " -o " ++ outputDir
   74     (exitCode, stdout, stdErr) <- readProcessWithExitCode "stack" ["exec", "--", "typus", "convert", inputDir, "-o", outputDir] ""
   75 
   76     case exitCode of
   77         ExitSuccess -> do
   78             putStrLn $ "Conversion completed successfully"
   79             putStrLn $ "Output: " ++ take 200 stdout
   80             return True
   81         ExitFailure code -> do
   82             hPutStrLn IO.stderr $ "typus convert failed with exit code " ++ show code
   83             hPutStrLn IO.stderr $ "STDOUT: " ++ take 500 stdout
   84             hPutStrLn IO.stderr $ "STDERR: " ++ take 500 stdErr
   85             return False

   87 -- Get list of typus files in directory
   88 getTypusFiles :: FilePath -> IO [FilePath]
   89 getTypusFiles dir = do
   90     allFiles <- listDirectory dir
   91     let typusFiles = filter (isSuffixOf ".typus") allFiles
   92     return $ map (dir </>) typusFiles

  106 -- Test running converted Go files
  107 testConvertedFiles :: FilePath -> [FilePath] -> IO [Bool]
  108 testConvertedFiles outputDir typusFiles = do
  109     outputExists <- doesDirectoryExist outputDir
  110     if not outputExists
  111         then do
  112             hPutStrLn IO.stderr $ "ERROR: Output directory " ++ outputDir ++ " does not exist"
  113             return []
  114         else do
  115             results <- forM typusFiles $ \typusFile -> do
  116                 let baseName = takeFileName typusFile
  117                 let goFile = outputDir </> dropExtension baseName <.> "go"
  118 
  119                 goExists <- doesFileExist goFile
  120                 if not goExists
  121                     then do
  122                         hPutStrLn IO.stderr $ "ERROR: Go file not found: " ++ goFile
  123                         return False
  124                     else do
  125                         putStrLn $ "Testing Go file: " ++ takeFileName goFile
  126 
  127                         -- Try to run Go file with timeout
  128                         (exitCode, stdout, stdErr) <- readProcessWithExitCode "timeout" ["10s", "go", "run", goFile] ""
  129                         let actualExitCode = if exitCode == ExitFailure 124
  130                                             then ExitFailure 1  -- Timeout
  131                                             else exitCode
  132 
  133                         case actualExitCode of
  134                             ExitSuccess -> do
  135                                 putStrLn $ "  ✓ Go file executed successfully"
  136                                 when (not $ null stdout) $ do
  137                                     putStrLn $ "    Output: " ++ take 200 stdout
  138                                     -- Check for expected output patterns
  139                                     checkExpectedOutput baseName stdout
  140                                 return True
  141                             ExitFailure _ -> do
  142                                 putStrLn $ "  ✗ Go file execution failed"

  144                                     putStrLn $ "    Error: " ++ take 200 stdErr
  145                                 return False
  146             return results

  148 -- Check if the output matches expected patterns for known files
  149 checkExpectedOutput :: FilePath -> String -> IO ()
  150 checkExpectedOutput filename output
  151     | "hello" `isInfixOf` filename =
  152         if "Hello" `isInfixOf` output
  153             then putStrLn $ "    ✓ Expected 'Hello' in output"
  154             else putStrLn $ "    ⚠ Expected 'Hello' in output"
  155     | "calculator" `isInfixOf` filename =
  156         if "10 + 20 = 30" `isInfixOf` output
  157             then putStrLn $ "    ✓ Expected calculator result in output"
  158             else putStrLn $ "    ⚠ Expected calculator result in output"
  159     | "algorithms" `isInfixOf` filename =
  160         if ("Original array:" `isInfixOf` output && "Sorted array:" `isInfixOf` output)
  161             then putStrLn $ "    ✓ Expected algorithm output patterns found"
  162             else putStrLn $ "    ⚠ Expected algorithm output patterns not found"
  163     | "data_structures" `isInfixOf` filename =
  164         if "Linked List contents:" `isInfixOf` output
  165             then putStrLn $ "    ✓ Expected data structures output found"
  166             else putStrLn $ "    ⚠ Expected data structures output not found"
  167     | otherwise = putStrLn $ "    (No specific output pattern to check for this file)"
